\section{Актуальность}
\begin{frame}{Актуальность}
    Языковые модели всё шире применяются как автономные агенты 
    в критических областях: медицина, финансы.
    
    \vspace{0.3cm}
    Поведение агентов формируется через \textbf{функцию вознаграждения} --- 
    числовую оценку качества их действий. 
    Однако любая функция вознаграждения --- лишь \textbf{приближение} 
    реальных целей человека.
    
    \vspace{0.3cm}
    \textbf{Захват вознаграждения} --- ситуация, при которой агент 
    находит способ максимизировать оценку, 
    \textbf{не выполняя} поставленную задачу.
    
\end{frame}

\section{Цель и задачи}
\begin{frame}{Цель и задачи}
    \textbf{Цель настоящего исследования:}
    \begin{itemize}
        \item[$\blacktriangleright$] Исследование устойчивости механизмов вознаграждения 
        в интеллектуальных агентах к смещению по длине ответа
    \end{itemize}
    
    \vspace{0.5cm}
    \textbf{Задачи:}
    \begin{enumerate}
        \item Провести анализ научно-технической литературы 
              и систематизировать известные уязвимости 
              механизмов вознаграждения
        \item Проанализировать модель вознаграждения 
              как источник уязвимости
        \item Разработать экспериментальный стенд 
              для проведения пентеста механизмов вознаграждения
        \item Провести пентест на моделях разного размера 
              и архитектуры
        \item Выявить и классифицировать паттерны деградации 
              при захвате вознаграждения
    \end{enumerate}
\end{frame}

\section{Обзор существующих решений}
\begin{frame}{Обзор существующих решений}
    Современные языковые модели могут проводить 
    \textbf{скрытые рассуждения} перед выдачей ответа:
    
    \vspace{0.3cm}
    \texttt{%
    <think>\\
    ~~~~Задача: 12 + 6 * 3 - 4 / 2\\
    ~~~~Сначала умножение: 6 * 3 = 18\\
    ~~~~Потом деление: 4 / 2 = 2\\
    ~~~~Итого: 12 + 18 - 2 = 28\\
    </think>\\
    Ответ: 28}
    
    \vspace{0.3cm}
    \begin{itemize}
        \item Содержимое \texttt{<think>} скрыто от пользователя
        \item Рассуждения повышают точность на сложных задачах
    \end{itemize}
    
    \vspace{0.3cm}
    \textbf{Примеры:} DeepSeek-R1, Qwen3
\end{frame}

\section{Выводы из обзора}
\begin{frame}{Выводы из обзора и требования к методологии}
    \textbf{Выводы из обзора:}
    \begin{enumerate}
        \item Смещение по длине --- системная проблема дообучения с подкреплением
        \item Влияние на модели с внутренними рассуждениями не исследовано
    \end{enumerate}
    
    \vspace{0.4cm}
    \textbf{Требования к методологии:}
    \begin{enumerate}
        \item Модель вознаграждения со смещением по длине
        \item Модели с механизмом внутренних рассуждений
        \item Варьирование размера модели и бюджета генерации
        \item Оценка по двум метрикам: вознаграждение и точность
    \end{enumerate}
\end{frame}

\section{Предложенное решение}
\begin{frame}{Предложенное решение}
    \textbf{Подход:} анализ устойчивости на основе методологии 
    сверхоптимизации вознаграждения (Gao и др., 2023)
    
    \vspace{0.3cm}
    \begin{enumerate}
        \item \textbf{Дообучение с подкреплением} ---
              оптимизация стратегии модели по сигналу от модели вознаграждения
        \item \textbf{Исследование моделей с внутренним этапом рассуждения} 
        \item \textbf{Два типа вознаграждения:}
              \begin{itemize}
                  \item Аппроксимация смещения по длине (функция насыщения)
                  \item Модель вознаграждения со штрафом за длину
              \end{itemize}
        \item \textbf{Периодическая оценка} --- каждые 100 шагов 
              измерение точности на расширенном бюджете генерации
    \end{enumerate}
\end{frame}

\section{Дизайн эксперимента}
\begin{frame}{Дизайн эксперимента}
\small

\textbf{Обновление стратегии:} используется алгоритм REINFORCE

\[
J(\theta)=\mathbb{E}_{y\sim \pi_\theta(\cdot|x)}[R(y)]
\]

\[
\nabla_\theta J = \mathbb{E}\left[(R - b)\,\nabla_\theta \log \pi_\theta(y|x)\right]
\]

\[
b \leftarrow 0.9\,b + 0.1\,\bar{R}
\]

\vspace{0.2cm}
\textbf{Обновление весов модели:} используется LoRA (низкоранговая адаптация)

Для матрицы весов $W \in \mathbb{R}^{d \times k}$ обновление задаётся как
\[
W' = W + \Delta W, \qquad \Delta W = BA
\]
где
\[
B \in \mathbb{R}^{d \times r}, \qquad A \in \mathbb{R}^{r \times k}, \qquad r \ll \min(d,k)
\]
\end{frame}

\section{Заключение}
\begin{frame}{Заключение}
    \small
    \textbf{Задача 1:} Анализ литературы --- смещение по длине 
    выявлено как основная уязвимость.
    
    \vspace{0.15cm}
    \textbf{Задача 2:} Подтверждено смещение в модели 
    вознаграждения.
    
    \vspace{0.15cm}
    \textbf{Задача 3:} Разработан экспериментальный стенд 
    для пентеста механизмов вознаграждения.
    
    \vspace{0.15cm}
    \textbf{Задача 4:} Пентест на 4 моделях, 2 архитектурах, 
    3 бюджетах. Захват при $\geq$1024 токенах на моделях 8B.
    
    \vspace{0.15cm}
    \textbf{Задача 5:} Классифицированы паттерны: 
    сжатие, удаление, агрессивный захват (точность 60\%), 
    скрытый захват (точность 80\%).
    
    \vspace{0.3cm}
    \textbf{Цель достигнута:} механизмы вознаграждения неустойчивы 
    к смещению по длине в моделях 8B при бюджете $\geq$1024 токенов.
\end{frame}
