\section{Актуальность}

\begin{frame}{Введение}
    LLM-агенты всё шире применяются для принятия решений 
    в критических областях: медицина, финансы, право, управление инфраструктурой.
    
    \vspace{0.4cm}
    Поведение агентов формируется через \textbf{функцию вознаграждения} --- 
    числовую оценку качества их действий. 
    Однако любая функция вознаграждения --- лишь \textbf{приближение} 
    реальных целей человека.
    
    \vspace{0.4cm}
    \textbf{Reward hacking} --- ситуация при которой агент находит способ 
    максимизировать вознаграждение, \textbf{не выполняя} поставленную задачу.
    
    \vspace{0.4cm}
    \textbf{Почему это опасно:}
    \begin{itemize}
        \item Все метрики показывают улучшение --- деградация \textbf{невидима}
        \item Проблема проявляется только при реальном использовании
    \end{itemize}
\end{frame}

\begin{frame}{REINFORCE: обновление политики по reward}
\small

\textbf{Цель:} увеличить ожидаемый reward модели
\[
J(\theta) = \mathbb{E}_{y \sim \pi_\theta(\cdot|x)}[R(y)]
\]

\vspace{0.2cm}
\textbf{Градиент REINFORCE:}
\[
\nabla_\theta J
=
\mathbb{E}\left[(R - b)\,\nabla_\theta \log \pi_\theta(y|x)\right]
\]

\vspace{0.2cm}
\textbf{Лог-вероятность ответа:}
\[
\log \pi_\theta(y|x)
=
\sum_{i=1}^{N}
\log \pi_\theta(t_i \mid t_1,\ldots,t_{i-1},x)
\]

\begin{itemize}
    \item $x$ --- входной запрос
    \item $y=(t_1,\dots,t_N)$ --- сгенерированный ответ
    \item $R$ --- reward для всего ответа
    \item $b$ --- baseline для снижения дисперсии:
    \[
    b \leftarrow 0.9\,b + 0.1\,\bar{R}
    \]
    \item $t_i$ --- $i$-й токен ответа, $N$ --- длина ответа
\end{itemize}

\vspace{0.2cm}
\textbf{Интерпретация:}
\begin{itemize}
    \item если $R>b$, вероятность такой траектории увеличивается
    \item если $R<b$, вероятность такой траектории уменьшается
    \item reward задаётся для всего ответа, а градиент распространяется на все его токены
\end{itemize}
\end{frame}

\section{Выводы из обзора}
\begin{frame}{Проблема RLHF}
\begin{itemize}
    \item Усиление языковых моделей с помощью обучения с подкреплением по человеческой обратной связи (\textbf{RLHF}) позволяет добиваться более управляемого и полезного поведения моделей.
    
    \item Однако даже при наличии человеческой оценки возможно возникновение побочных эффектов, в частности \textbf{захвата вознаграждения} (\textbf{reward hacking}), при котором модель оптимизирует не то поведение, которое от неё ожидается.

    \item Одним из наиболее распространённых проявлений такого эффекта является \textbf{искусственное удлинение ответа}: модель начинает генерировать более длинные ответы не потому, что они содержат больше полезной информации, а потому, что большая длина начинает коррелировать с более высокой оценкой со стороны reward-модели.
\end{itemize}
\end{frame}

\begin{frame}{Предлагаемое исследование}
\begin{itemize}
    \item Предлагается исследовать, \textbf{подвержены ли reasoning-модели эффекту reward hacking}, проявляющемуся в \textbf{увеличении длины ответа}.

    \item Помимо самого факта наличия данного эффекта, планируется изучить, \textbf{как именно reasoning-модели ведут себя в таком сценарии}.

    \item Результатом исследования должно стать лучшее понимание того, характерен ли этот вид хакинга для reasoning-моделей и в каких формах он проявляется.
\end{itemize}
\end{frame}

\begin{frame}{Метод в экспериментах: REINFORCE + LoRA}
\small

\textbf{Обновление политики:} используется REINFORCE-style policy gradient
\[
\nabla_\theta J = \mathbb{E}\left[(R - b)\,\nabla_\theta \log \pi_\theta(y|x)\right]
\]

\vspace{0.3cm}
\textbf{LoRA: low-rank adaptation}

Для матрицы весов $W \in \mathbb{R}^{d \times k}$ обновление задаётся как
\[
W' = W + \Delta W, \qquad \Delta W = BA
\]
где
\[
B \in \mathbb{R}^{d \times r}, \qquad A \in \mathbb{R}^{r \times k}, \qquad r \ll \min(d,k)
\]

\begin{itemize}
    \item исходные веса $W$ замораживаются
    \item обучаются только малые матрицы $A$ и $B$
    \item $BA$ задаёт low-rank поправку к исходным весам
    \item число обучаемых параметров резко уменьшается
    \item снижаются требования к памяти и вычислениям
\end{itemize}


\end{frame}

\begin{frame}{Выводы}
    \begin{enumerate}
        \item \textbf{Reasoning-модели уязвимы к reward hacking.}
              При оптимизации с RM, имеющей length bias,
              Qwen3-8B удаляет \texttt{<think>} и теряет 30\% accuracy,
              при этом RM Score растёт.

        \item \textbf{Бюджет генерации --- критический параметр безопасности.}
              512 токенов --- модель адаптируется без потери качества.
              1024 токена --- модель эксплуатирует length bias.

        \item \textbf{Три паттерна деградации:}
              сжатие reasoning, удаление reasoning, reward hacking
              (неправильные ответы + повторения).
    \end{enumerate}
\end{frame}
